from subprocess import Popen, PIPE

def run_sql_script(conn_str, filename):
    sqlplus = Popen(['sqlplus', '-S', conn_str], stdin=PIPE, stdout=PIPE, stderr=PIPE)
    sqlplus.stdin.write('@'+filename)
    return sqlplus.communicate()
